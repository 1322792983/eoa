CREATE DATABASE EOA;
USE EOA;
CREATE TABLE TB_ROLE(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	ROLE_NAME VARCHAR(20) NOT NULL UNIQUE COMMENT '角色名称',
	ROLE_KIND INT COMMENT '角色级别',
	ROLE_INFO VARCHAR(100) COMMENT '角色简介',
	ACTIVE_FLAG INT ,
	CREATE_BY VARCHAR (10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME	
);

CREATE TABLE TB_CLAZZ(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	CLAZZ_NAME VARCHAR(20) NOT NULL UNIQUE COMMENT '班级名称',
	STU_COUNT INT COMMENT '学生数量',
	CLAZZ_INFO VARCHAR(100) COMMENT '班级简介',
	ACTIVE_FLAG INT ,
	CREATE_BY VARCHAR (10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME	
);

CREATE TABLE TB_STUDENT(
	ROW_ID BIGINT PRIMARY KEY AUTO_INCREMENT COMMENT '主键',
	CLAZZ_ID BIGINT COMMENT '班级主键',
	STU_CODE VARCHAR(20) NOT NULL UNIQUE COMMENT '学生编号',
	STU_NAME VARCHAR(20) NOT NULL COMMENT '学生名称',
	STU_SEX INT NOT NULL COMMENT '学生性别',
	STU_BIRTHDAY DATETIME NOT NULL COMMENT '学生生日',
	ACTIVE_FLAG INT ,
	CREATE_BY VARCHAR (10),
	CREATE_DATE DATETIME,
	UPDATE_BY VARCHAR(10),
	UPDATE_DATE DATETIME,
	FOREIGN KEY (CLAZZ_ID) REFERENCES `TB_CLAZZ` (ROW_ID)
);
